package family;
class FamilyOverload {
    void introduce() {
        System.out.println("This is our family.");
    }
    void introduce(String father) {
        System.out.println("Father: " + father);
    }
    void introduce(String father, String mother) {
        System.out.println("Father: " + father + ", Mother: " + mother);
    }
    public static void main(String[] args) {
        FamilyOverload f = new FamilyOverload();
        f.introduce();
        f.introduce("Ramesh");
        f.introduce("Ramesh", "Sita");
    }
}
